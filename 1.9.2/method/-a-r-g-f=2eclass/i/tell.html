ARGFが現在開いているファイルのファイルポインタの現在の位置を整数で返します。 

ARGV.replace(
  Dir.glob("*").reject{|name| FileTest.file?(name) == false}
)

while c = ARGF.getc
  printf("%d '%c' %s\n", ARGF.pos, c, ARGF.filename)
  if ARGF.pos == 3
    ARGF.skip
  end
end
