instance method Hash#sort

sort -> Array
sort {|a, b| ... } -> Array

ハッシュを [key, value] を要素とする配列の配列に変換して，それをソー トした配列を返します。 

h = { "a" => 20, "b" => 30, "c" => 10  }
h.sort                       #=> [["a", 20], ["b", 30], ["c", 10]]
h.sort {|a,b| a[1]<=>b[1]}   #=> [["c", 10], ["a", 20], ["b", 30]]
